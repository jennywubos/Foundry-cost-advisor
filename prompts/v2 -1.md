## COST CALCULATION — FULLY GENERATED

```javascript
function calcMonthlyCost(monthlyInputM, monthlyOutputM, cacheRate, tier) {
  if (tier === 'ptu') {
    const ptus = calcRecommendedPTUs(monthlyInputM, monthlyOutputM);
    return ptus * PRICING.ptu.perPTUperMonth;
  }
  const p = PRICING[tier];
  const cachedM   = monthlyInputM * cacheRate;
  const uncachedM = monthlyInputM * (1 - cacheRate);
  return (uncachedM * p.inputPer1M)
       + (cachedM   * p.cachedInputPer1M)
       + (monthlyOutputM * p.outputPer1M);
}

function calcRecommendedPTUs(monthlyInputM, monthlyOutputM) {
  // Convert monthly totals → average TPM (720 hrs/month = 24×7×30)
  // For demand sizing, use peak TPM from mock data directly instead
  // This function uses monthly averages as a conservative baseline
  const avgInputTPM  = (monthlyInputM  * 1e6) / (720 * 60);
  const avgOutputTPM = (monthlyOutputM * 1e6) / (720 * 60);
  // Effective input TPM: output counts as 4× input toward PTU utilization
  // Source: https://learn.microsoft.com/en-us/azure/foundry/openai/how-to/provisioned-throughput-onboarding
  const effectiveTPM = avgInputTPM + (avgOutputTPM * PRICING.ptu.outputRatio);
  const base = effectiveTPM / PRICING.ptu.tpmPerPTU_input;
  return Math.ceil(base * 1.17); // 17% headroom buffer (standard practice)
}

function calc12MonthProjection(baseInputM, baseOutputM, cacheRate) {
  const ptus = calcRecommendedPTUs(baseInputM, baseOutputM);
  return Array.from({length: 12}, (_, i) => {
    const g = Math.pow(1.10, i); // 10% MoM compound growth
    return {
      month: i + 1,
      standard: calcMonthlyCost(baseInputM * g, baseOutputM * g, cacheRate, 'standard'),
      priority: calcMonthlyCost(baseInputM * g, baseOutputM * g, cacheRate, 'priority'),
      ptuMonthly: ptus * PRICING.ptu.perPTUperMonth,          // flat
      ptuAnnual:  ptus * PRICING.ptu.perPTUperMonth * 0.85,   // flat, -15%
    };
  });
}
```

```javascript
function generateRecommendation(userData, data) {
  const throttleHigh  = data.throttledCallsPct > 0.05;
  const throttleMod   = data.throttledCallsPct > 0.01;
  const quotaHigh     = (data.quotaAllocatedTPM / data.quotaLimitTPM) > 0.50;
  const cacheLow      = data.cacheMatchRate < 0.10;
  const workloadSpiky = data.workloadPattern === 'spiky';
  const workload247   = data.workloadPattern === '247';

  // Score — derive from data, not hardcoded
  let priorityScore = 0, ptuScore = 0;
  if (workloadSpiky)   priorityScore += 3;
  if (throttleHigh)    priorityScore += 3;
  if (!quotaHigh)      priorityScore += 1;
  if (cacheLow && workloadSpiky) priorityScore += 1;
  if (workload247)     ptuScore += 3;
  if (quotaHigh)       ptuScore += 3;
  if (throttleMod)     ptuScore += 1;
  if (!workloadSpiky)  ptuScore += 1;

  const rec = ptuScore > priorityScore ? 'ptu' : 'priority';
  const bullets = [];
  let headline = '';
  const throttlePct   = (data.throttledCallsPct * 100).toFixed(1);
  const quotaPct      = ((data.quotaAllocatedTPM / data.quotaLimitTPM) * 100).toFixed(0);
  const cachePct      = (data.cacheMatchRate * 100).toFixed(0);
  const ptus          = calcRecommendedPTUs(data.monthlyInputTokensM, data.monthlyOutputTokensM);

  if (rec === 'priority') {
    headline = `Your usage pattern — ${throttlePct}% throttle rate during business hours with variable overnight volume — matches the Priority Processing design target: "${OFFER_CONTEXT.priorityProcessing.designTarget}"`;

    bullets.push(
      `${throttlePct}% throttled calls (30-day avg) → Standard is blocking users at peak. ` +
      `${OFFER_CONTEXT.priorityProcessing.quotaBehavior}`
    );
    bullets.push(
      `Spiky workload (~${data.workingHoursPerMonth} business hrs/month) → PTU dedicated capacity ` +
      `would sit unused overnight and on weekends. Priority Processing is PAYG — cost scales with actual consumption.`
    );
    if (cacheLow) bullets.push(
      `${cachePct}% cache match rate — consistent with Global Standard behavior: ` +
      `"${OFFER_CONTEXT.cacheOnGlobalStandard.warning}" Each legal contract is unique, ` +
      `further reducing prefix reuse. PTU cache economics require >40% hit rate to justify the capacity cost.`
    );
    bullets.push(
      `⚠️ Preview caveat: ${OFFER_CONTEXT.priorityProcessing.previewStatus} ` +
      `${OFFER_CONTEXT.priorityProcessing.latencyTarget}`
    );
    bullets.push(
      `${OFFER_CONTEXT.priorityProcessing.downgradeRisk} ` +
      `Monitor via Azure Monitor → split by ServiceTierRequest/ServiceTierResponse.`
    );

  } else {
    headline = `Your consistent 24/7 workload at ${quotaPct}% quota utilization matches the PTU design target: "${OFFER_CONTEXT.ptu.designTarget}"`;

    if (quotaHigh) bullets.push(
      `${quotaPct}% quota utilization → approaching the Standard TPM rate limit. ` +
      `PTU provides dedicated capacity outside the shared quota pool, eliminating rate-limit risk.`
    );
    bullets.push(
      `24/7 consistent load → PTU dedicated GPU capacity is fully utilized around the clock. ` +
      `${OFFER_CONTEXT.ptu.sla}`
    );
    if (data.cacheMatchRate === 0) bullets.push(
      `Current cache match rate: 0% on Global Standard → with PTU, opportunity to reach 40–50% ` +
      `cache hit rate for FAQ-heavy support prompts. ${OFFER_CONTEXT.ptu.cacheBenefit} ` +
      `Projected throughput: ~${data.projectedRPMWithCache} RPM on same ${ptus} PTUs. ` +
      `(Source: Chris Hoder, Azure AI Foundry Ignite 2025 demo)`
    );
    bullets.push(
      `Foundry PTU calculator → ${ptus} PTUs for ${(data.peakInputTPM/1e6).toFixed(2)}M peak input TPM. ` +
      `Fixed cost: $${(ptus * PRICING.ptu.perPTUperMonth).toLocaleString()}/month — ` +
      `not exposed to token price growth as volume scales.`
    );
  }

  return {
    rec,
    headline,
    bullets,
    sources: OFFER_CONTEXT[rec === 'ptu' ? 'ptu' : 'priorityProcessing'].sources,
  };
}
```

## SPARKLINE GENERATORS

```javascript
function generateHistory(data) {
  if (data.workloadPattern === 'spiky') {
    return Array.from({length: 60}, (_, i) => {
      const busy = (i >= 5 && i <= 22) || (i >= 35 && i <= 52);
      const spike = i === 14 || i === 48;
      const rpm = busy
        ? data.peakRPM * (0.82 + Math.random() * 0.22)
        : data.peakRPM * (0.04 + Math.random() * 0.08);
      return {
        min: i, rpm: Math.round(rpm),
        inputTPM: Math.round(rpm * data.avgPromptTokens),
        outputTPM: Math.round(rpm * data.avgCompletionTokens),
        throttlePct: spike ? 0.35 : (busy ? 0.06 + Math.random()*0.08 : 0.001),
        cacheRate: data.cacheMatchRate + (Math.random()-0.5)*0.005,
      };
    });
  } else {
    return Array.from({length: 60}, (_, i) => {
      const rpm = data.peakRPM * (0.90 + Math.random() * 0.12);
      return {
        min: i, rpm: Math.round(rpm),
        inputTPM: Math.round(rpm * data.avgPromptTokens),
        outputTPM: Math.round(rpm * data.avgCompletionTokens),
        throttlePct: data.throttledCallsPct * (0.7 + Math.random() * 0.6),
        cacheRate: data.cacheMatchRate,
      };
    });
  }
}
```
